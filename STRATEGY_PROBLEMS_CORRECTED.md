```java
// 在开仓前检查价格位置
BigDecimal[] bb = calculateBollingerBands(klines, 20, 2);
BigDecimal upperBand = bb[0];
BigDecimal lowerBand = bb[2];
BigDecimal currentPrice = klines.get(0).getClosePrice();

// 做多检查：价格不能接近上轨
if (signal.isBuy()) {
    BigDecimal upperThreshold = upperBand.multiply(new BigDecimal("0.95"));
    if (currentPrice.compareTo(upperThreshold) > 0) {
        log.warn("⚠️ 价格${}接近上轨${}, 暂不做多", currentPrice, upperBand);
        return;
    }
}

// 做空检查：价格不能接近下轨
if (signal.isSell()) {
    BigDecimal lowerThreshold = lowerBand.multiply(new BigDecimal("1.05"));
    if (currentPrice.compareTo(lowerThreshold) < 0) {
        log.warn("⚠️ 价格${}接近下轨${}, 暂不做空", currentPrice, lowerBand);
        return;
    }
}
```

**效果**：
- 避免$4595+做多（接近上轨$4597）
