public boolean validateOptimalSignal(TradingSignal signal, MarketContext context) {
    Double adx = context.getIndicator("ADX");
    Double williamsR = context.getIndicator("WilliamsR");
    
    if (adx == null || williamsR == null) {
        return false;
    }
    
    // 规则1：只交易ADX≥30的强趋势
    if (adx < 30.0) {
        log.warn("⛔ ADX={} < 30，不是强趋势，拒绝交易", adx);
        return false;
    }
    
    // 规则2：做多 - Williams R在-80~-60黄金区间
    if (signal.getType() == SignalType.BUY) {
        if (williamsR > -60.0 || williamsR < -80.0) {
            log.warn("⛔ 做多WR={}不在黄金区间-80~-60，拒绝", williamsR);
            return false;
        }
        log.info("🎯 最优做多组合！ADX={}, WR={}", adx, williamsR);
        return true;
    }
    
    // 规则3：做空 - Williams R在-60~-20区间（严格限制）
    if (signal.getType() == SignalType.SELL) {
        if (williamsR < -60.0 || williamsR > -20.0) {
            log.warn("🚫 做空WR={}不在安全区间-60~-20，拒绝", williamsR);
            return false;
        }
        
        // 额外验证：做空需要超强信号（因为历史表现差）
        if (signal.getStrength() < 80) {
            log.warn("🚫 做空信号强度{}不足（需要≥80），拒绝", signal.getStrength());
            return false;
        }
        
        log.warn("⚡ 做空条件勉强满足：ADX={}, WR={}, 强度={}", 
                adx, williamsR, signal.getStrength());
        return true;
    }
    
    return false;
}
